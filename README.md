## Install

```bash
npm install --save redux-geogebra
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'redux-geogebra'
import 'redux-geogebra/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
